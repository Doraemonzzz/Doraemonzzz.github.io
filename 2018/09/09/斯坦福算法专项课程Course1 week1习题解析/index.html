编程题

实现Karatsuba乘法：

def Karatsuba(x, y):
    if x < 10 or y < 10:
        return x * y
    else:
        #计算位数
        k = min(len(str(x)), len(str(y))) // 2
        u = 10 ** k
        a = x // u
        b = x - a * u
        c = y // u
        d = y - c * u
        #递归
        ac = Karatsuba(a, c)
        bd = Karatsuba(b, d)
        abcd = Karatsuba(a + b, c + d)
        #合并结果
        adbc = abcd - ac - bd
        
        res = u ** 2 * ac + u * adbc + bd
        
        return res
